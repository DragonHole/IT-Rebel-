Fix X-VPN will renew it with 500 MB. Note: you may need to unlock first, and you need the latest X-VPN installed." buttons {"Full Unlock", "Fix X-VPN"})		set confirm to display dialog "Are your sure to " & answer & "?" buttons {"Maybe later", "Yes"}		if answer = "Fix X-VPN" and button returned of confirm = "Yes" then		tell application "X-VPN" to quit		if (exists ((path to home folder as string) & "Library:Application Support:kmg")) then			tell application "Finder" to delete ((path to home folder as string) & "Library:Application Support:kmg")		end if		tell application "X-VPN" to activate	end if		if answer = "Full Unlock" and button returned of confirm = "Yes" then		# start deleting files 				do shell script "rm -f ~/Library/Extensions/SophosFileMonitor.kext"		do shell script "rm -f ~/Library/Extensions/SophosFileProection.kext"		do shell script "rm -f ~/Library/Extensions/SophosPortInterceptor.kext"		do shell script "rm -f ~/Library/Extensions/SophosWebProtection.kext"		do shell script "rm -f ~/Library/Frameworks/SophosGenericsCommon.framework"		do shell script "rm -f ~/Library/Frameworks/SophosGenericsCore.framework"		do shell script "rm -f ~/Library/LaunchAgents/com.sophos.endpoint.uiserver.plist"		do shell script "rm -f ~/Library/LaunchAgents/com.sophos.agent.plist"		do shell script "rm -f ~/Library/LaunchAgents/com.jamfsoftware.jamf.agent.plist"		do shell script "rm -f ~/Library/LaunchDaemons/com.sophos.common.servicemanager.plist"		do shell script "rm -f ~/Library/LaunchDaemons/com.jamfsoftware.task.1.plist"		do shell script "rm -f ~/Library/LaunchDaemons/com.jamfsoftware.startupItem.plist"		do shell script "rm -f ~/Library/LaunchDaemons/com.jamfsoftware.jamf.daemon.plist"		do shell script "rm -f ~/Library/LaunchDaemons/com.jamf.management.daemon.plist"				tell application "Terminal"			do script with command "sudo /usr/local/bin/jamf removeFramework"		end tell				display alert "you are now free"	end if	end tell
